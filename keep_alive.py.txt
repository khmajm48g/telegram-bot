from flask import Flask
from threading import Thread

app = Flask('')

@app.route('/')
def home():
    return "I'm alive! 😎"

def run():
    print("🚀 Starting Flask server on port 8080...")  # رسالة تأكيد تشغيل السيرفر
    app.run(host="0.0.0.0", port=8080)

def keep_alive():
    t = Thread(target=run)
    t.start()
